class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        idx = 0
        zerocounter = 0
        while idx < len(nums):
            if nums[idx] == 0:
                del nums[idx]
                zerocounter += 1
                continue
            idx += 1
        nums += [0]*zerocounter
